Logging

Logging

package main

import (
    "log"
    "os"
)

var logger *log.Logger

func init() {
    logApp()
}

// log
func logApp() {
    file, err := os.OpenFile("app.log", os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0666)
    if err != nil {
        log.Fatalln("Failed to open log file", err)
    }
    logger = log.New(file, "INFO ", log.Ldate|log.Ltime|log.Lshortfile)
}

func _Danger(args ...interface{}) {
    logger.SetPrefix("ERROR ")
    logger.Println(args...)
}

func _Warning(args ...interface{}) {
    logger.SetPrefix("WARNING ")
    logger.Println(args...)
}

func main() {
    // something happened that it's important to log
    _Danger("problem...")
}

In code above create the following entry into the file:

ERROR 2018/02/14 22:11:56 main.go:25: problem...
